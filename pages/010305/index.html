源码位于： src/mixins/initDict.js，代码如下

（1）引入组件

import initDict from '@/mixins/initDict'
export default {
  mixins: [initDict]
}

（2）使用钩子函数获取字典

import initDict from '@/mixins/initDict'
export default {
 mixins: [initDict],
 created() {
    this.$nextTick(() => {
      // 加载数据字典
      this.getDict('job_status')
    })
  }
}
